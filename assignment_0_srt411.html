funt1 = function(vect)
  {
  for (i in 1:length(vect))
    {
    if (i < 5 | i > 90)
      {
      vect[i] = vect[i] * 10
      }else{
      vect[i] = vect[i] * 0.1 
      }
    }
  return (vect) 
  }
